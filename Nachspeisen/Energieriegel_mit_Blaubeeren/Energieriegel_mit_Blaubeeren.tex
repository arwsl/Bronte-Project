    [
        preparationtime = {\SI{10}{\minute}},
        bakingtime = {\SI{20}{\minute} bis \SI{25}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{topbottomheat=\SI{180}{\celsius}}},
        portion = {10 Riegel},
        source = {myprotein.com}
    ]
    {Energie-Riegel mit \addtoidx{Blaubeeren}}

    \graph
        {
            big=Nachspeisen/Energieriegel_mit_Blaubeeren/big.png,
            small=Nachspeisen/Energieriegel_mit_Blaubeeren/small.png
        }

    \introduction{Diese Energie-Riegel mit Blaubeeren liefern den heißersehnten Energie-Kick, ohne schwer im Magen zu liegen.}

    \ingredients{
        \multicolumn{2}{l}{\textbf{Boden}} \\
        1 & Apfel, geraspelt \\
        2 & Bananen \\
        \SI{200}{\g} & \addtoidx{Haferflocken} \\
        \SI{1}{\EL} & \addtoidx{Proteinpulver} \\
        \SI{3}{\EL} & Ahornsirup \\
        \SI{3}{\EL} & Mandelbutter \\
        \\
        \multicolumn{2}{l}{\textbf{obere Schicht}} \\
        \SI{150}{\g} & Blaubeeren \\
        \SI{3}{\EL} & Kürbiskerne \\
        \SI{3}{\EL} & Sonnenblumenkerne \\
        \SI{80}{\g} & Haferflocken \\
        \SI{5}{\EL} & Ahornsirup \\
        \\
        \multicolumn{2}{l}{\textbf{optionales Topping}} \\
        & veganes Karamell
    }

    \preparation{
        \step Heize den Backofen auf \SI{180}{\celsius} vor. Zuerst wird der Boden vorbereitet. In einer Küchenmaschine (oder Mixer) vermengst du zunächst alle Zutaten für den Boden. Die Masse sollte nicht fein püriert sein.
        \step Lege den Boden einer \SI{20}{\cm}x\SI{20}{\cm}-Backform mit Backpapier aus. Die Seiten pinselst du mit etwas Öl ein. Backe den Boden für 10 Minuten.
        \step Nimm dann alles aus dem Ofen und stelle es anschließend kurz beiseite. Um die obere Schicht zuzubereiten, vermengst du in einer Schüssel alle Zutaten für die Schicht und verteilst die Masse dann auf dem Boden. Backe die Riegel nun für weitere \SI{10}{\minute}-\SI{15}{\minute}. Nimm die Backform aus dem Ofen und lasse die Energie-Riegel abkühlen. Schneide sie am Ende nur noch in die passende Form und genieße einen kleinen Energie-Boost.
    }
