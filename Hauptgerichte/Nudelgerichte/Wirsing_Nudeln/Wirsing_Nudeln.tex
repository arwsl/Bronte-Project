\begin{recipeDP}
    [
        preparationtime = {\SI{40}{\minute}},
        % bakingtime = {\SI{ZEIT}{\minute} bis \SI{ZEIT}{\minute}},
        % bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{TEMPERATUR}{\celsius}}},
        portion = {4 Portionen},
        source = {zuckerjagdwurst.com}
    ]
    {One Pot Pasta mit Wirsing und Hack}

    \graph
        {
            big=Hauptgerichte/Nudelgerichte/Wirsing_Nudeln/big.jpg,
            small=Hauptgerichte/Nudelgerichte/Wirsing_Nudeln/small.jpg
        }

    % \introduction{
    %     Schneller veganer Nudeleintopf mit Wirsing, veganem Hack und Paprika -- alles in einem Topf zubereitet.
    % }

    \ingredients{
        1 & Zwiebel \\
        1 & Knoblauchzehe \\
        \SI{500}{\g} & \addtoidx{Wirsing} \\
        1 & rote Paprika \\
        1 & gelbe Paprika \\
        \SI{500}{\g} & veganes \addtoidx{Hack} \\
        \SI{1}{\TL} & Paprikapulver, edelsüß \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Muskat \\
        \SI{3}{\EL} & Tomaten-Gemüse-Mark \\
        \SI{1000}{\ml} & Gemüsebrühe \\
        \SI{250}{\ml} & passierte Tomaten \\
        \SI{250}{\g} & Nudeln \\
        \SI{2}{\EL} & Hefeflocken \\
        \SI{300}{\g} & vegane Crème fraîche \\
        & Pflanzenöl \\
        & Petersilie \\
        & Salz \\
        & Pfeffer
    }

    \preparation{
        \step Zwiebel und Knoblauch schälen und fein würfeln.
        Wirsing in feine Streifen schneiden und beide Paprika würfeln.
        \step Pflanzenöl in einem großen Topf bei mittlerer Hitze erwärmen.
        Zwiebelwürfel dazugeben und etwa 2 bis 3 Minuten anschwitzen, bis sie glasig sind.
        Danach Knoblauch dazugeben und etwa 2 Minuten weiterbraten.
        Dann auch das vegane Hack, Paprikapulver, Muskat und Tomaten-Gemüse-Mark mit anbraten, bis alles gut gebräunt ist.
        \step Die Paprikawürfel hinzugeben und kurz anbraten.
        Dann alles mit Gemüsebrühe und passierten Tomaten aufgießen, die Nudeln hinzugeben und kochen, bis die Nudeln al dente sind.
        Etwa 2 bis 3 Minuten vor der angegebenen Kochzeit den Wirsing hinzugeben und unterrühren.
        \step Zum Schluss mit Hefeflocken, Paprikapulver, Muskat, Salz und Pfeffer abschmecken und mit veganer Creme und frischer Petersilie servieren.
    }

    % \suggestion[TITEL EINES VORSCHLAGS]{
	% 	VORSCHLAG (DURCH HORIZONTALE LINIE VOM REZEPT GETRENNT)
    % }

    \hint{
        Statt frischem Hack können auch Linsen oder andere Hülsenfrüchte verwendet werden.
    }

\end{recipeDP}
