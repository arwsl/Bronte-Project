\begin{recipeDP}
    [
        preparationtime = {\SI{40}{\minute}},
        % bakingtime = {\SI{ZEIT}{\minute} bis \SI{ZEIT}{\minute}},
        % bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{TEMPERATUR}{\celsius}}},
        portion = {4 Portionen},
        source = {@elavegan}
    ]
    {Cremige \addtoidx{Nudelsuppe}}

    \graph
        {
            big=Hauptgerichte/Nudelgerichte/Cremige_Nudelsuppe/big.jpg,
            small=Hauptgerichte/Nudelgerichte/Cremige_Nudelsuppe/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Suppe}} \\
        \SI{225}{\g} & Möhren \\
        \SI{75}{\g} & Stangensellerie \index{Sellerie} \\
        \SI{200}{\g} & Brokkoli \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\EL} & Öl \\
        1 & Zwiebel \\
        4 & Knoblauchzehen \\
        \SI{1300}{\ml} & Gemüsebrühe \\
        \SI{2}{\TL} & Zwiebelpulver \\
        \SI{2}{\TL} & italienische Kräuter \\
        \SI{1}{\TL} & Liebstöckel \\
         & Salz \\
         & Pfeffer \\
        \SI{250}{\g} & Nudeln \\
        \\
        \multicolumn{2}{l}{\textbf{Cashew-Bohnen-Creme}} \\
		\SI{400}{\g} & weiße (Cannellini-) Bohnen \index{Bohnen!weiße} \\
        \SI{75}{\g} & \addtoidx{Cashews} \\
        \SI{360}{\ml} & Pflanzenmilch \\
        \SI{1}{\EL} & Zitronensaft \\
        \SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & Salz
    }

    \preparation{
        \step Die Cashews ein paar Stunden einweichen oder für \SI{20}{\minute} in Wasser köcheln lassen, bis sie weich sind. Sobald die Cashews weich sind, zusammen mit den Cannellini-Bohnen und der pflanzlichen Milch in einen Mixer geben und bei hoher Geschwindigkeit für etwa zwei Minuten pürieren oder bis die Creme ganz glatt, samtig und cremig ist. Dann Beiseite stellen.
        \step Für die Suppe: Öl in einem großen Topf bei mittlerer Hitze erhitzen, Zwiebel und Gemüse dazugeben und ca. 4 Minuten anbraten. Dann den Knoblauch und die Gewürze hinzugeben und eine weitere Minute anbraten.
        \step Nun die Gemüsebrühe dazugeben. Zum Kochen bringen und dann die Hitze reduzieren. Die Cashew-Bohnen-Creme dazugeben und umrühren. Die Suppe etwa 6 bis 8 Minuten köcheln lassen.
        \step Zum Schluss Nudeln nach Wahl hinzufügen und die Suppe 10 Minuten weiterköcheln lassen, bis die Nudeln gar sind. Zwischendurch immer wieder umrühren. Wenn die Suppe zu dick ist, entweder mehr Gemüsebrühe oder pflanzliche Milch hinzufügen!
    }
