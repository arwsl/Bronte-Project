\begin{recipeDP}
    [
        preparationtime = {\SI{45}{\minute}},
        % bakingtime = {\SI{ZEIT}{\minute} bis \SI{ZEIT}{\minute}},
        % bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{TEMPERATUR}{\celsius}}},
        portion = {4 Portionen},
        source = {elavegan.com}
    ]
    {Nudeln mit Kichererbsen}

    \graph
        {
            big=Hauptgerichte/Nudelgerichte/Nudeln_mit_Kichererbsen/big.jpg,
            small=Hauptgerichte/Nudelgerichte/Nudeln_mit_Kichererbsen/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{2}{\EL} & Öl \\
        \SI{1}{\TL} & Oregano \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Thymian \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Rosmarin \\
        \SI{1}{\TL} & Salz \\
         & Chiliflocken \\
        \SI{150}{\g} & Zwiebeln \\
        \SI{120}{\g} & Staudensellerie\index{Sellerie} \\
        \SI{200}{\g} & Möhren \\
        3 & Knoblauchzehen \\
        \SI{50}{\g} & Tomatenmark \\
        \SI{540}{\g} & \addtoidx{Kicherebsen} aus der Dose \\
        \SI{1200}{\ml} & Gemüsebrühe \\
        \SI{250}{\g} & trockene Nudeln \\
         & Petersilie
    }

    \preparation{
        \step Zuerst die Zwiebel, den Sellerie und die Möhren klein würfeln.
        Den Knoblauch fein hacken.
        Öl in einem großen Topf bei mittlerer Hitze erhitzen, dann Oregano, Thymian, Rosmarin, Salz, Pfeffer und rote Chiliflocken hinzufügen.
        30 Sekunden lang anbraten, dann die Zwiebel, den Sellerie und die Karotten hinzufügen.
        Etwa 5 Minuten unter gelegentlichem Rühren braten, dann den Knoblauch hinzufügen und eine weitere Minute mitbraten.
        \step Den Topf mit etwas von der Gemüsebrühe ablöschen und das Tomatenmark hinzugeben.
        Etwa die Hälfte der Kichererbsen hinzufügen (den Rest für später aufheben) und mit der Gemüsebrühe aufgießen.
        Zum Kochen bringen, dann die Hitze reduzieren und die Nudeln hinzufügen.
        \step Bei schwacher bis mittlerer Hitze köcheln lassen, bis die Nudeln al dente sind (die Angaben auf der Packung lesen und die Kochzeit um eine Minute reduzieren).
        \step Die restlichen Kichererbsen in einen Mixer geben und 3-4 Schöpfkellen der Gemüsebrühe hinzufügen.
        So lange pürieren, bis alles glatt ist.
        Diese Mischung zurück in den Topf geben und umrühren.
        In Schüsseln servieren, mit frischen Kräutern garnieren

    }
