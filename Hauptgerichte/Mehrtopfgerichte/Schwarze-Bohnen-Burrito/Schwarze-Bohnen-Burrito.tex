\begin{recipeDP}
    [
        preparationtime = {\SI{10}{\minute}},
        bakingtime = {\SI{45}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{200}{\celsius}}},
        portion = {10 Burritos},
        source = {plantyou.com}
    ]
    {Schwarze-Bohnen-Burrito}

    \graph
        {
            big=Hauptgerichte/Mehrtopfgerichte/Schwarze-Bohnen-Burrito/big.jpg,
            small=Hauptgerichte/Mehrtopfgerichte/Schwarze-Bohnen-Burrito/small.jpg
        }

    % \introduction{
    %     Vegane Burritos mit gerösteten Kartoffeln, schwarzen Bohnen, Paprika und cremiger Chipotle-Sauce – perfekt für Meal-Prep und gefrierschrankfreundlich.
    % }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Kartoffeln}} \\
        3 & Paprika (rot, orange, gelb) \\
        1 & Zwiebel \\
        1 & Tomate \\
        6 bis 8 & \addtoidx{Kartoffeln} \\
        \SI{1}{\EL} & Paprikapulver \\
        \SI{1}{\EL} & Kreuzkümmel \\
        & Salz und Pfeffer \\
        \\
        \multicolumn{2}{l}{\textbf{Chipotle Schwarze Bohnen}} \\
        3 & Knoblauchzehen \\
        1 & \addtoidx{Chipotle-Chili} \\
        \SI{2}{\TL} & Kreuzkümmel \\
        \SI{1}{\TL} & Paprikapulver \\
        \SI{1}{\EL} & Hefeflocken \\
        \SI{1}{\TL} & Salz \\
        2 Dosen & schwarze Bohnen\index{Bohnen!schwarz} (je 540 g) \\
        \SI{60}{\ml} & Gemüsebrühe \\
        \\
        \multicolumn{2}{l}{\textbf{Chipotle-Sauce}} \\
        \SI{180}{\ml} & Cashews \\
        \SI{120}{\ml} & pflanzliche Milch \\
        \SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & Meersalz \\
        \SI{1}{\TL} & Ahornsirup \\
        1 & Limette \\
        2 & Chipotle-Chili \\
        \SI{1}{\TL} & Adobo-Sauce \\
        \\
        10 & \addtoidx{Tortilla-Wraps} \\
        & Koriander
    }

    \preparation{
        \step Den Ofen auf \SI{200}{\celsius} (Umluft) vorheizen.
        Paprika, Zwiebel, Tomate und geschälte Kartoffeln würfeln.
        Auf einem Backblech das gewürfelte Gemüse mit Paprikapulver, Kreuzkümmel, Salz und Pfeffer mischen.
        Verteilen und 45 Minuten rösten, bis die Kartoffeln knusprig sind.
        \step Währenddessen die Knoblauchzehen würfeln oder fein hacken.
        Die Chipotle-Chili in Scheiben schneiden.
        In einer Pfanne bei mittlerer Hitze den Knoblauch, Chipotle-Chili, Kreuzkümmel, Paprikapulver, Hefeflocken, Salz und einen Esslöffel Öl anbraten.
        Anschwitzen bis es duftet.
        \step Die schwarzen Bohnen abtropfen lassen und abspülen.
        Die 2 Dosen schwarze Bohnen und Gemüsebrühe zur Pfanne hinzufügen.
        Umrühren bis alles vermischt ist. Mit einem Kartoffelstampfer die Bohnen zu einer dicken Paste zerdrücken.
        Dabei sollten noch einige Bohnen intakt bleiben für die Textur.
        \step Die Cashews über Nacht in Wasser einweichen (alternativ 15 Minuten mit heißem Wasser übergießen).
        Für die Chipotle-Sauce die eingeweichten Cashews, pflanzliche Milch, Meersalz, Ahornsirup, den Saft der Limette und 2 Chipotle-Chili mit 1 TL Adobo-Sauce in einen Hochleistungsmixer geben und zu einer glatten Sauce mixen.
        \step Den Koriander grob hacken.
        Die Burritos zusammenbauen: Die Wraps einzeln auslegen und mit zwei gehäuften Löffeln der Kartoffelmischung und einem gehäuften Löffel der schwarzen Bohnen füllen.
        Mit Chipotle-Sauce und gehacktem Koriander abschließen und einwickeln.
    }

    % \suggestion[TITEL EINES VORSCHLAGS]{
	% 	VORSCHLAG (DURCH HORIZONTALE LINIE VOM REZEPT GETRENNT)

    \hint{
        Im Kühlschrank bis zu 4 Tage lagern oder in Folie oder Butterbrotpapier einwickeln und in einen Gefrierbeutel geben für bis zu 3 Monate.
        In der Heißluftfritteuse, Mikrowelle oder in der Pfanne auf dem Herd aufwärmen.
    }

\end{recipeDP}
