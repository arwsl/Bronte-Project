        1 & Zucchini \\
        \SI{150}{\ml} & Pflanzendrink \\
        \SI{2}{\EL} & Olivenöl \\
        \SI{120}{\g} & Mehl \\
        \SI{2}{\EL} & Semmelbrösel \\
        \SI{1}{\TL} & Backpulver \\
        \SI{1}{\TL} & Salz \\
        1 Prise & Pfeffer
     }

    \preparation{
        \step Die Zucchini reiben und soviel wie möglich von dem Wasser ausdrücken. Dann mit den restlichen Zutaten vermixen und gut mit Salz und Pfeffer abschmecken.
        \step Nun je 1 EL von dem Teig in eurem heißen Waffeleisen für etwa \SI{4}{\minute} backen.
    }
