\begin{recipeDP}
    [
        preparationtime = {\SI{15}{\minute}},
        bakingtime = {\SI{30}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{180}{\celsius}}},
        portion = {4 Portionen},
        source = {healthygirlkitchen.com}
    ]
    {Gebackene \addtoidx{Spaghetti}}

    \graph
        {
            big=Hauptgerichte/Ofengerichte/Gebackene_Spaghetti/big.jpg,
            small=Hauptgerichte/Ofengerichte/Gebackene_Spaghetti/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{500}{\g} & Spaghetti (dünn) \\
        \SI{100}{\g} & Grünkohl \\
        1 Glas & sonnengetrocknete Tomaten\index{Tomaten!sonnengetrocknet} \\
        1 & Zwiebel \\
        1 & \addtoidx{Zucchini} \\
        \SI{530}{\ml} & Tomatensoße \\
        \SI{590}{\ml} & Wasser \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Knoblauchpulver \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Pfeffer \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Basilikum \\
         & Olivenöl
    }

    \preparation{
        \step Zuerst den Ofen auf \SI{180}{\celsius} vorheizen. Die dünnen Spaghetti in eine große Auflaufform geben und mit etwas Olivenöl beträufeln. Das Öl gleichmäßig verteilen.
        \step Die sonnengetrockneten Tomaten klein schneiden, die Zwiebel und de Zucchini würfeln und dann alles auf die Nudeln geben. Das Wasser und die Tomatensoße (alternativ: passierte Tomaten) darüber geben und alles gleichmäßig verteilen. Die Nudeln müssen alle mit Flüssigkeit bedeckt sein. Abschließend mit Salz, Knoblauch, Pfeffer und Basilikum würzen.
        \step Die Nudeln dann für \SI{30}{\minute} backen, bis das Wasser von den Nudeln aufgenommen ist.
    }

    % \suggestion[TITEL EINES VORSCHLAGS]{
	% 	VORSCHLAG (DURCH HORIZONTALE LINIE VOM REZEPT GETRENNT)
    % }

    \hint{
        Werden statt dünner Spaghetti andere Nudeln verwendet, müssen etwa \SI{750}{\ml} Wasser verwendet werden. Die Backzeit verlängert sich um etwa \SI{15}{\minute}.
    }

\end{recipeDP}
