\begin{recipeDP}
    [
        preparationtime = {\SI{20}{\minute}},
        bakingtime = {\SI{35}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{topbottomheat=\SI{175}{\celsius}}},
        portion = {4 Portionen},
        source = {@elavegan}
    ]
    {\addtoidx{Fittata}}

    \graph
        {
            big=Hauptgerichte/Ofengerichte/Frittata/big.jpg,
            small=Hauptgerichte/Ofengerichte/Frittata/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{1}{\EL} & Öl \\
        \SI{100}{\g} & Zwiebeln \\
        \SI{120}{\g} & Paprika \\
        \SI{120}{\g} & \addtoidx{Süßkartoffeln} \\
        \SI{80}{\g} & \addtoidx{Zucchini} \\
        \SI{80}{\g} & Tomaten \\
        \SI{75}{\g} & Erbsen \\
        2 & Knoblauchzehen \\
        \\
        \multicolumn{2}{l}{\textbf{Teig}} \\
        \SI{240}{\ml} & Gemüsebrühe \\
        \SI{150}{\g} & Kichererbsenmehl \index{Kichererbsen!-mehl} \\
        \SI{30}{\g} & Speisestärke \\
        \SI{20}{\g} & Hefeflocken \\
        \SI{1}{\TL} & Backpulver \\
        \SI{1}{\TL} & Salz \\
		\SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & Kala Namak \\
        \SI[parse-numbers = false]{\nicefrac{3}{4}}{\TL} & italienische Gewürzmischung \\\SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Pfeffer \\
        \SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & Räucherpaprika \\
        \SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & Kreuzkümmel \\
    }

    \preparation{
        \step Öl in einer ofenfesten Pfanne erhitzen. Auch die Innenseiten der Pfanne einfetten. Zwiebel, Paprika und Süßkartoffel hinzugeben. Bei mittlerer Hitze \ca 5 bis 6 Minuten unter gelegentlichem Rühren braten. Zucchini, Tomaten, Erbsen und Knoblauch hinzufügen. Weitere 3 bis 4 Minuten braten, dann die Herdplatte abstellen.
        \step Den Ofen auf \SI{175}{\celsius} vorheizen. Diesen Schritt überspringen, falls die Frittata auf dem Herd zubereitet wird.
        \step Der Teig wird in einer mittleren bis großen Schüssel zubereitet. Kichererbsenmehl, Speisestärke, Hefeflocken, Backpulver und alle Gewürze in die Schüssel geben. Mit einem Schneebesen gut vermischen. Dann das Wasser in 3 Portionen unter ständigem Rühren hinzufügen, bis keine Klumpen mehr vorhanden sind.
        \step Den Teig nun in die Pfanne geben und umrühren, um das Gemüse damit zu bedecken. 30 bis 35 Minuten im Ofen backen. Die Ränder der Frittata sollten goldbraun sein und es sollten mehrere Risse erkennbar sein.
    }

    \suggestion{
		Anstatt sie im Ofen zu Backen, lässt sich die Frittata auch in einer großen Pfanne mit Deckel zubereiten: Die gleichen Schritte wie oben befolgen, dann in eine Pfanne geben und mit einem Deckel abdecken und bei niedriger bis mittlerer Hitze \ca 12 bis 15 Minuten braten. Mit Hilfe eines Tellers vorsichtig wenden und weitere 10 Minuten braten. Beim Wenden die Pfanne einölen.
    }

    % \hint{
    %     HINWEIS (IN EINEM KASTEN UNTEN AUF DER SEITE)
    % }

\end{recipeDP}
