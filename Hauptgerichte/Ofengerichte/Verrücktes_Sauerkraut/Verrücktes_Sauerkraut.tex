\begin{recipeDP}
    [
        preparationtime = {\SI{25}{\minute}},
        bakingtime = {\SI{15}{\minute} bis \SI{25}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{200}{\celsius}}},
        portion = {6 Portionen},
        source = {@bambergleon}
    ]
    {Verrücktes Sauerkraut}

    \graph
        {
            big=Hauptgerichte/Ofengerichte/Verrücktes_Sauerkraut/big.jpg,
            small=Hauptgerichte/Ofengerichte/Verrücktes_Sauerkraut/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{2}{\EL} & Öl \\
        2 & Zwiebeln \\
        \SI{1500}{\g} & (Wein-)Sauerkraut \index{Sauerkraut} \\
        \SI{1}{\TL} & Gemüsebrühe (Pulver) \\
        \SI{500}{\g} & veganes Hack \\
        \SI{150}{\g} & Frischkäse \\
        1 Tube & Tomatenmark \\
        \SI{200}{\g} & Streukäse
    }

    \preparation{
        \step Die Zwiebeln gewürfelt in dem Öl anschwitzen.
        Wenn diese glasig sind, das Sauerkraut hinzugeben und mit der Gemüsebrühe (Pulver) abschmecken.
        \step Das vegane Hack nach 5 Minuten mit hinein geben und auf mittlerer Hitze garen, dabei gut rühen und das Hack zerteilen, damit es rundum gar wird.
        den Frischkäse und das Tomatenmark mit hineingaben und gut vermischen.
        Zusammen etwa zehn Minuten braten.
        \step Alles in eine leicht gefettete Auflaufform geben, mit Streukäse bedecken und bei \SI{200}{\celsius} für etwa eine viertel Stunde backen (bis der Käse gut zerlaufen ist).
    }

\end{recipeDP}
