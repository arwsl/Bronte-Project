\begin{recipeDP}
    [
        preparationtime = {\SI{20}{\minute}},
        bakingtime = {\SI{30}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{180}{\celsius}}},
        portion = {20 x 25 cm Backform},
        source = {zuckerjagdwurst.com}
    ]
    {Brookies}

    \graph
        {
            big=Gebaeck/Kuchen/Brookies/big.jpg,
            small=Gebaeck/Kuchen/Brookies/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Brownieteig}\index{Brownie}} \\
        \SI{200}{\g} & Weizenmehl (Type 405) \\
        \SI{200}{\g} & Zucker \\
        \SI{8}{\g} & Vanillezucker \\
        \SI{50}{\g} & Kakaopulver \\
        \SI{200}{\g} & Zartbitterschokolade \\
        \SI{100}{\g} & \addtoidx{Apfel!-mus} \\
        \SI{100}{\ml} & Sonnenblumenöl \\
        \SI{200}{\ml} & Sojamilch \\
        \\
        \multicolumn{2}{l}{\textbf{Cookieteig}\index{Cookie}} \\
        \SI{200}{\g} & Weizenmehl (Type 405) \\
        \SI{80}{\g} & Zucker \\
        \SI{125}{\g} & Margarine \\
        \SI{8}{\g} & Vanillezucker \\
        \SI{1}{\TL} & Backpulver \\
        1 Prise & Salz \\
        \SI{2}{\EL} & Apfelmus \\
        \SI{150}{\g} & Zartbitter-Schokoladentropfen \\
    }

    \preparation{
        \step Für den Brownie-Teig Weizenmehl, Zucker, Vanillezucker und Kakaopulver in einer Schüssel vermischen.
        Die Zartbitterschokolade im Wasserbad schmelzen.
        \step Anschließend die geschmolzene Schokolade, Apfelmus, Pflanzenöl und pflanzliche Milch zu den trockenen Zutaten geben, gut unterrühren und in eine eingefettete Form (20 x 25 cm) geben.
        \step Den Backofen auf \SI{180}{\celsius} (Umluft) vorheizen.
        Für den Cookie-Teig Weizenmehl, Zucker, vegane Butter, Vanillezucker, Backpulver, Salz und Apfelmus in einer Schüssel vermengen, bis sich eine große Kugel formen lässt.
        Die Schokotropfen ebenfalls untermischen.
        \step Den Cookie-Teig auf dem Brownie-Teig in Fetzen verteilen und für 30 Minuten im Ofen auf mittlerer Schiene backen.
        Nach 30 Minuten den Ofen ausschalten und den Brookie im Ofen bei geschlossener Tür abkühlen lassen.
        Sobald die Brookies abgekühlt sind in Stücke schneiden und servieren.
    }
