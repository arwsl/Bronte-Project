\begin{recipeDP}
    [
        preparationtime = {\SI{15}{\minute}},
        bakingtime = {\SI{45}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{topbottomheat=\SI{180}{\celsius}}},
        portion = {1 Kastenform (25 cm)},
        source = {heyfoodsister.de}
    ]
    {Apfel-Walnuss-Kuchen}

    \graph
        {
            big=Gebaeck/Kuchen/Apfel-Walnuss-Kuchen/big.jpg,
            small=Gebaeck/Kuchen/Apfel-Walnuss-Kuchen/small.jpg
        }

    % \introduction{
    %     Einfacher veganer Rührkuchen mit Apfelstücken und Walnüssen, überzogen mit Zimtglasur -- in 5 Minuten angerührt und herrlich saftig.
    % }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Kuchen}} \\
        \SI{400}{\g} & \addtoidx{Mehl} \\
        \SI{125}{\g} & brauner Zucker\index{Zucker!braun} \\
        1 \pck & Vanillezucker \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{} & Bio-Zitrone (Abrieb) \\
        1 \pck & Backpulver \\
        \SI{150}{\ml} & Öl \\
        \SI{175}{\ml} & Hafermilch \\
        \SI{150}{\g} & \addtoidx{Apfelmus} \\
        \SI[parse-numbers = false]{\nicefrac{1}{4}}{\TL} & \addtoidx{Zimt} \\
        2 & feste Äpfel \\
        \SI{100}{\g} & \addtoidx{Walnüsse} \\
        \\
        \multicolumn{2}{l}{\textbf{Guss}} \\
        \SI{125}{\g} & Puderzucker \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{} & Zitrone (Saft) \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Zimt
    }

    \preparation{
        \step Backofen auf \SI{180}{\celsius} Ober-/Unterhitze vorheizen.
        Eine Kastenform (25 cm) mit Backpapier auslegen oder gründlich einfetten.
        Äpfel schälen und in Stückchen schneiden, mit Zimt mischen.
        Walnüsse grob hacken.
        \step In einer großen Schüssel Mehl, Zucker, Vanillezucker, Backpulver, Salz und Zitronenabrieb vermengen.
        In separater Schüssel Öl, Hafermilch und Apfelmus verrühren.
        Eine Mulde in der Mehlmischung formen und die flüssigen Zutaten hineingießen.
        Mit einem Schneebesen nur so lange verrühren, bis der Teig zusammenkommt.
        Äpfel und Walnüsse unterheben.
        \step Teig in die Form füllen und etwa 45 Minuten backen; Stäbchenprobe machen.
        Kuchen kurz in der Form und anschließend auf einem Kuchengitter vollständig auskühlen lassen.
        Für den Guss Puderzucker und Zimt mischen.
        Zitronensaft löffelweise unterrühren, bis die Glasur die gewünschte Konsistenz hat.
        Auf dem erkalteten Kuchen verteilen.
    }

    % \suggestion[TITEL EINES VORSCHLAGS]{
	% 	VORSCHLAG (DURCH HORIZONTALE LINIE VOM REZEPT GETRENNT)
    % }

    \hint{
        Bei Rührkuchen gilt: Bloß nicht zu viel rühren, sonst wird der Teig zäh. Die Eier werden durch Apfelmus ersetzt -- passt perfekt zu den Apfelstücken und Walnüssen. Optional nur mit Puderzucker bestäuben oder mit weißer Schokoglasur glasieren.
    }

\end{recipeDP}
