\begin{recipeDP}
    [
        preparationtime = {\SI{30}{\minute}},
        bakingtime = {\SI{45}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{topbottomheat=\SI{180}{\celsius}}},
        portion = {1 Blech (30 x 40 cm)},
        source = {eat-this.org}
    ]
    {Apfelkuchen von Blech}

    \graph
        {
            big=Gebaeck/Kuchen/Apfelkuchen_vom_Blech/big.jpg,
            small=Gebaeck/Kuchen/Apfelkuchen_vom_Blech/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Rührteig}} \\
        \SI{450}{\ml} & Sojamilch \\
        1 & Zitrone \\
        \SI{450}{\g} & Dinkelmehl \index{Mehl!Dinkel-} \\
        \SI{50}{\g} & Speisestärke \\
        \SI{150}{\g} & Zucker \\
        \SI{17}{\g} & Backpulver \\
        \SI{1}{\TL} & Natron \\
        1 Prise & Salz \\
        \SI{1}{\pck} & Vanillezucker \\
        \SI{200}{\ml} & Rapsöl \\
        \\
        \multicolumn{2}{l}{\textbf{Belag}} \\
        \SI{600}{\g} & Äpfel \index{Apfel} \\
        \SI{3}{\EL} & Aprikosenmarmelade \index{Marmelade!Aprikosen-}
    }

    \preparation{
        \step Für den Rührteig Sojamilch und \SI{50}{\ml} Zitronensaft vermengen und beiseitestellen.
        Mehl, Speisestärke, Zucker, Backpulver, Natron und Salz vermischen.
        Die Sojamilch-Mischung, Vanilleextrakt, Zitronenzesten und Rapsöl dazugeben und rasch zu einem glatten Teig verrühren.
        Backofen auf \SI{180}{\celsius} Ober-/Unterhitze vorheizen.
        \step Äpfel in \SI[parse-numbers = false]{\nicefrac{1}{2}}{\cm} dünne Spalten schneiden und in einer Schüssel mit übrigem Zitronensaft vermengen.
        Teig auf ein mit Backpapier ausgelegtes Backblech gießen und gleichmäßig verteilen.
        Die Apfelspalten fächerartig in Reihen auf den Teig legen und leicht andrücken. Auf mittlerer Schiene 45 bis \SI{50}{\minute} backen. Stäbchenprobe machen.
        \step Nach dem Backen noch warm mit leicht angewärmter Aprikosenmarmelade bestreichen und komplett abkühlen lassen.
    }

    % \suggestion[TITEL EINES VORSCHLAGS]{
	% 	VORSCHLAG (DURCH HORIZONTALE LINIE VOM REZEPT GETRENNT)
    % }

    \hint{
        Für die Äpfel eignet sich am besten eine säuerliche Sorte wie Boskop oder Elstar.
    }

\end{recipeDP}
