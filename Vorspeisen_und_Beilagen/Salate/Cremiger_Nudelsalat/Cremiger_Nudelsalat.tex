\begin{recipeDP}
    [
        preparationtime = {\SI{25}{\minute}},
        % bakingtime = {\SI{ZEIT}{\minute} bis \SI{ZEIT}{\minute}},
        % bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{TEMPERATUR}{\celsius}}},
        portion = {4 Portionen},
        source = {@elavegan}
    ]
    {Cremiger \addtoidx{Nudelsalat}}

    \graph
        {
            big=Vorspeisen_und_Beilagen/Salate/Cremiger_Nudelsalat/big.jpg,
            small=Vorspeisen_und_Beilagen/Salate/Cremiger_Nudelsalat/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{250}{\g} & \addtoidx{Nudeln} \\
        1 & grüne Paprika \index{Paprika!grün} \\
        1 & Zwiebel \\
        4 & Tomaten \\
        \SI{200}{\g} & \addtoidx{Gurke} \\
        \SI{150}{\g} & Essiggurken \index{Gurke!Essig-}\\
        & Oliven \\
        \\
        \multicolumn{2}{l}{\textbf{Dressing}} \\
        \SI{260}{\g} & weiße Bohnen \index{Bohnen!weiße} \\
        \SI{160}{\g} & Pflanzenmilch \\
        \SI{40}{\g} & Cashews \\
        2 & Knoblauchzehen \\
        \SI{1}{\EL} & Senf \\
        \SI{1}{\EL} & Limettensaft \\
        \SI{1}{\EL} & Apfelessig \\
		\SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz \\
         & Pfeffer \\
         & Dill
    }

    \preparation{
        \step Nudeln gemäß der Anleitung auf der Packung kochen. Und die Cashews in heißem Wasser einweichen. Weiße Bohnen, pflanzliche Milch, die eingeweichten Cashewnüsse, Knoblauchzehen, Senf, Limettensaft, Essig, Meersalz und schwarzen Pfeffer in einen Mixer geben. Auf höchster Stufe pürieren bis das Dressing vollständig glatt und cremig ist. Dann Beiseite stellen. Das Gemüse klein schneiden.
        \step In einer großen Schüssel die abgekühlten Nudeln, Paprikaschote, Zwiebel, Tomaten, Gurke und die Dillgurken vermengen. Darüber das Dressing gießen und umrühren. Mit frischen Kräutern garnieren und servieren.

        \vspace{3.5cm}
    }

    
    \suggestion[Cremiger Kartoffelsalat]{
		\index{Kartoffelsalat}Anstatt der Nudeln lassen sich auch \SI{900}{\g} festkochende Kartoffeln verwenden. Die dafür in mittelgroße Würfel schneiden und in Salzwasser für etwa \SI{10}{\minute} köcheln lassen. Dann mit dem Gemüse und dem Dressing zu einem Salat vermengen.
    }

    % \hint{
    %     HINWEIS (IN EINEM KASTEN UNTEN AUF DER SEITE)
    % }

\end{recipeDP}
