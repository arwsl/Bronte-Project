    \ingredients{
        \SI{500}{\g} & \addtoidx{Nudeln} \\
        1 Glas & rotes \addtoidx{Pesto} \\
        \SI{180}{\g} & sonnengetrocknete Tomaten\index{Tomaten!sonnengetrocknet} \\
        \SI{125}{\g} & \addtoidx{Rucola} \\
        \SI{150}{\g} & Hirtenkäse \\
    }

    \preparation{
        \step Zuerst die Nudeln nach Packungsanleitung kochen und dann das Glas Pesto mit den Nudeln in eine Schale geben. Wenn möglich etwas abkühlen lassen. Währenddessen den Rucola waschen und grob zerreißen.
        \step Die getrockneten Tomaten auch grob zerkleinern und den Hirtenkäse würfeln. Dann alle Zutaten zusammen vermengen.
    }
