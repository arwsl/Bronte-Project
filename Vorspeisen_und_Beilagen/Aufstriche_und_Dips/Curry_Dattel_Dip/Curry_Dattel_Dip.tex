\begin{recipeDP}
    [
        preparationtime = {\SI{20}{\minute}},
        % bakingtime = {\SI{ZEIT}{\minute} bis \SI{ZEIT}{\minute}},
        % bakingtemperature = {\protect\bakingtemperature{fanoven=\SI{TEMPERATUR}{\celsius}}},
        % portion = {PORTIONEN/MENGE},
        source = {cookidoo.de}
    ]
    {Curry-Dattel-Dip}

    \graph
        {
            big=Vorspeisen_und_Beilagen/Aufstriche_und_Dips/Curry_Dattel_Dip/big.jpg
            % small=TEIL/KAPITEL/REZEPT/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{150}{\g} & \addtoidx{Datteln}, entsteint \\
        1 & Knoblauchzehe \\
        \SI{350}{\g} & \addtoidx{Frischkäse} \\
        \SI{150}{\g} & vegane Creme \\
        \SI{2}{\TL} & Currypulver \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz \\
        1 Prise & Cayennepfeffer
    }

    \preparation{
        \step die Datteln in kleine Stücke schneiden und in einer kleinen Schüssel mit heißem Wasser übergießen. 
        Für etwa \SI{10}{\minute} einweichen lassen.
        \step Anschließend die eingeweichten Datteln abgießen, das Wasser auffangen und zusammen mit den restlichen Zutaten in einen Standmixer geben.
        Alles gut pürieren und zwischendurch immer wieder mit einem Löffel nach unten schieben, bis eine cremige Masse entsteht.
        Abhängig von der Konsistenz muss noch etwas vom Dattelwasser hinzugefügt werden.
    }
