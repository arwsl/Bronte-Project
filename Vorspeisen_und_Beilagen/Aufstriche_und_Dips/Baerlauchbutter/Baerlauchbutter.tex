    [

    {Bärlauchbutter}

    \graph
        {
            big=Vorspeisen_und_Beilagen/Aufstriche_und_Dips/Baerlauchbutter/big.jpg,
            small=Vorspeisen_und_Beilagen/Aufstriche_und_Dips/Baerlauchbutter/small.jpg
        }

    % \introduction{
    %     EINLEITUNG
    % }

    \ingredients{
        \SI{250}{\g} & Butter (z. B. Alsan) \\
        \SI{50}{\g} & Bärlauch \\
        \SI{2}{\EL} & Zitronensaft \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz
    }

    \preparation{
        \step Die vegane Butter frühzeitig aus dem Kühlschrank nehmen.
        Bei Zimmertemperatur weich werden lassen.
        \step Die Bärlauchblätter (ohne Stiele) gründlich waschen und trocknen.
        Die Blätter klein schneiden oder hacken.
        In einer Schüssel alle Zutaten miteinander verrühren.
        Dafür eignet sich eine Gabel gut.
        Am Ende in eine Form oder Dose geben und im Kühlschrank wieder fest werden lassen.
    }
