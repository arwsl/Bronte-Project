\begin{recipeDP}
    [
        preparationtime = {\SI{20}{\minute}},
        bakingtime = {\SI{15}{\minute}},
        bakingtemperature = {\protect\bakingtemperature{topbottomheat=\SI{200}{\celsius}}},
        portion = {18 Stück},
        source = {cutiekulla.com}
    ]
    {Datteln im Nicht-Speckmantel}

    \graph
        {
            big=Nachspeisen_und_Snacks/Snacks/Datteln_im_Nichtspeckmantel/big.jpg,
            small=Nachspeisen_und_Snacks/Snacks/Datteln_im_Nichtspeckmantel/small.jpg
        }

    \introduction{
        Vegane Datteln im Speckmantel mit knusprigem Reispapier-Mantel – perfekt als Fingerfood für Feiern, Vorspeise oder edler Snack.
    }

    \ingredients{
        \multicolumn{2}{l}{\textbf{Marinade}} \\
        \SI{1}{\TL} & Sojasauce \\
        \SI{1}{\TL} & Öl \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Liquid Smoke \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Ahornsirup \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Räucherpaprika \\
        \SI[parse-numbers = false]{\nicefrac{1}{2}}{\TL} & Salz \\
        1 Msp & Knoblauchgranulat \\
        \\
        \multicolumn{2}{l}{\textbf{Füllung}} \\
        3 & Reisblätter (rechteckig)\index{Reispapier} \\
        18 & \addtoidx{Datteln} \\
        18 & Mandeln, geräuchert oder gesalzen
    }

    \preparation{
        \step Den Backofen auf \SI{200}{\celsius} Ober-/Unterhitze vorheizen.
        Das Reispapier in 6 gleich breite Streifen schneiden (ca. 2,5 cm).
        Alle flüssigen Zutaten und Gewürze zu einer Marinade verrühren.
        Die Datteln gegebenenfalls entsteinen und mit je einer Mandel (oder anderen Nüssen wie z.B. Pistazien) füllen.
        \step Nun die Reispapierstreifen Stück für Stück, für ca. 10 bis 15 Sekunden, in eine Schüssel Wasser tauchen und auf ein (mit Backpapier ausgelegtes) Backblech legen.
        Die feuchten Reispapierstreifen durch die Marinade ziehen oder noch besser: von beiden Seiten mit der Marinade einpinseln und um eine Dattel wickeln und mit der Nahtstelle nach unten aufs Backblech legen.
        \step Nun ca. 15 Minuten backen und anschließend abkühlen lassen.
    }
